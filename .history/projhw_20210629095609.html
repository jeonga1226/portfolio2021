function loadChallengeCard(){
    gbStrgGoalObj = SessUtil.getGoalList();
 
    /* 걷기왕 */
    PAGE_DATA.todayGoalStep  = gbStrgGoalObj.DAILY_STP; 
    $('#bGoalStepCnt').text(comma.set(PAGE_DATA.todayGoalStep));
 
    /* 나를 이기다 */
    PAGE_DATA.monthGoalStep = gbStrgGoalObj.MONTHLY_STP;    // 목표(지난달 누적 걸음수)
    var monNumDay = 0 ;              // 지난 달 일수
    var monthRcmndPer = 0;              // 권장걸음 비율 
    
    // 권장걸음 = (지난 달 누적 걸음수/지난 달 일수) * 진행일 수 -> (지난달 누적 걸음수/이번달 일수)*현재일수)
    var date = new Date();
    var endDayOfMonth = new Date( date.getFullYear(), date.getMonth() + 1 , 0); // 이번달거
    monNumDay = Number(moment(endDayOfMonth).format('DD'));
    PAGE_DATA.monthRecmndStep = Math.round(( PAGE_DATA.monthGoalStep / monNumDay) * Number(moment().format('DD')));
    
    monthRcmndPer = parseInt((PAGE_DATA.monthRecmndStep / PAGE_DATA.monthGoalStep) * 100);
    monthRcmndPer = (isNaN(monthRcmndPer)) ? 0 : monthRcmndPer;
 
    $('#spMonWinGoalCnt').html('<strong>지난 달 누적걸음</strong> ('+ comma.set(PAGE_DATA.monthGoalStep) +'걸음)' ); 
    $('#bMonWinRecmndCnt').text(comma.set(PAGE_DATA.monthRecmndStep) ); 
    $('#spMonWinBarRcmnd').attr('style','width : '+ monthRcmndPer +'%');
 
    /* 체중 조절 */
    var storgActData = SessUtil.getActivityData();
    var lastMealObj = (storgActData.meal == undefined) ? {} : storgActData.meal;            // 스토리지 내 마지막 식단 정보 
    var lastWeightObj = (storgActData.weight == undefined) ? {} : storgActData.weight.last; // 스토리지 내 체중 정보
 
    PAGE_DATA.todayWeightCnt = 0;
 
    if(lastMealObj.endDate == moment().format('YYYYMMDD')){
        var todaMealArr = lastMealObj.perType.slice(0,2);   // 길이 최대 2
        $.each(todaMealArr, function(idx, val){
            $('#spWeiMeal > i').eq(idx).removeClass("dim");
        });
        
        PAGE_DATA.todayWeightCnt += todaMealArr.length;
    }
 
    if(lastWeightObj.date == moment().format('YYYY.M.D')){
        $('#spWeiWeight > i').removeClass('dim');
        PAGE_DATA.todayWeightCnt += 1;
    }
 
    // retrieveMovementBowelState(); // 추후 오픈 
    drawBadgePercentage("all");
    PAGE_DATA.stepCardType = "daily";
    
}

}

// 실시간 동기화하여 총걸음수 가져오기 
var samIntervalId;
function intervalTotalStep(param, func){
    if(isEmpty(param)){     // 삼성, 구글 
        window[func]("CBexWNSHealthTodayTotalStep");
        if (samIntervalId == undefined) {
            samIntervalId = setInterval(function(){
                window[func]("CBexWNSHealthTodayTotalStep");
            }, PAGE_DATA.INTERVAL_TIME);
        }
    }else{                  //iOS
        window[func](param,"CBexWNSHealthTodayTotalStep");
        if (samIntervalId == undefined) {
            samIntervalId = setInterval(function(){
                window[func](param,"CBexWNSHealthTodayTotalStep");
            }, PAGE_DATA.INTERVAL_TIME);
        }
    }
}

//삼성헬스 - 활동량(걸음수) 가져오기 삼성활동콜백
function CBGetSamsungActivity(resultJsonString) {
    // console.log("CBGetSamsungActivity");
    // console.log('resultJsonString = ' + resultJsonString);
    var resultJsonObj = JSON.parse(resultJsonString);
 
    if(resultJsonObj.code == 0) {
        // 성공. 비지니스 로직 수행
 
        var activityRawDatas = [];      
        var activityData = resultJsonObj.resultData;
 
            if(!isEmpty(activityData)){
            
            // 걸음수로 거리 계산하기 long step 걸음, int height 키...
            // @TEST: ApiUtil.getDistanceByStep(2369, 182);
            // 걸음수로 칼로리 계산하기 double weight 무게, float dis 걸음수
            // @TEST: ApiUtil.getCalorie(69.2, 1742);
            var activityRawDatas = {
                entries : [],
                source : {
                    mode : 20,
                    type : 1009,
                    name : "hello",
                    product: {
                        name : "samsungHealth",
                        vender : "samsung"
                    }
                }
            }
            
            $.each(activityData, function(key, value){
                var fromTime = moment(value.start_time).format("YYYY-MM-DD HH:mm:ss");
                var toTime = moment(value.end_time).format("YYYY-MM-DD HH:mm:ss");
                var steps = value.count;       // 걸음수
                var distance = value.distance; // m
                var kcalorie = value.calorie;  // kcal
                var distanceVal = distance/1000;
                var calorieVal = kcalorie;
                
                activityRawDatas.entries.push({
                        distance : {
                            value : distanceVal,
                            unit : "km"
                        },
                        calories : {
                            value : calorieVal,
                            unit : "kcal"
                        },
                        steps : steps,
                        period : {
                            from : fromTime,
                            to : toTime
                        }
                });
            });
            
            MNet.httpSend({
                path : INTF.ACT.ACT102,
                data : {
                    rawData : activityRawDatas
                },
                indicator : false,
                callback : function(rd){
                    // 글로벌에 지난달 누적 걸음수 저장
                    if(rd.monthlyGoalStep != undefined && rd.monthlyGoalStep != 0){
                        var goalObj = {
                            "DAILY_STP" : rd.dailyGoalStep,
                            "MONTHLY_STP" : rd.monthlyGoalStep
                        }
                        SessUtil.setGoalList(goalObj);   // 목표 설정 
 
                        if(rd.activityCode != undefined){
                            var sessionUser =  SessUtil.getUserInfo();
                            sessionUser.activityNm = rd.activityNm;
                            sessionUser.activityCode = rd.activityCode;
 
                            SessUtil.updateUserInfo(sessionUser);
                        }                       
                    }
 
                    MData.global("lastActivityDate", moment(rd.lastActivity).format('YYYY-MM-DD HH:mm:ss'));
                    getMonthlyStep();
                }
            });
        }else{
            // 데이터 없음 
            getMonthlyStep();
        }
        
    } else {
        // 실패. 에러처리등...
        exWNSHealthDisconnect('CBSamsungDisconnect');
        getMonthlyStep();
    }
}

function checkIOSChildActivityData( o ) {

function CBexWNGetIOSHourlyStepCount(jsonStr) {
    var node = JSON.parse(jsonStr);
    
    $.each(node.resultData.stepList, function(i, o){
        
        var fromTime = moment(o.time, 'YYYY-MM-DD HH:mm').format('YYYY-MM-DD HH:mm:ss');
        var toTime = moment(o.time, 'YYYY-MM-DD HH:mm').format('YYYY-MM-DD HH:59:59');
        var dis = ApiUtil.getDistanceByStep(o.step, IOHealthCheckHeight);
        var kcal = ApiUtil.getCalorie(IOHealthCheckWeight, dis);
        var distanceVal = parseFloat(dis) / 1000; //거리 m => mk
        var kcalorieVal = parseFloat(kcal); //칼로리 cal => kcal
        var rawData = {
                distance : {
                    value : distanceVal,
                    unit : "km"
                },
                calories : {
                    value : kcalorieVal,
                    unit : "kcal"
                },
                steps : o.step,
                period : {
                    from : fromTime,
                    to : toTime
                }
        };
 
        IOHealthAppDatas.entries.push(rawData);
    });
    IOHealthCheckIndex = IOHealthCheckIndex + 1;
    CBcheckIOHealthAppData();
}
 
/**
 * 다음 프로세스를 위한 체크
 * 1. 인덱스(+1)와 length 가 같아지면 최종 전송
 * 2. rawData 쌓였으면 전송하고 rawData 초기화 한뒤 index 이어서 계속 조회
 * 3. 인덱스가 남아있으면 index 이어서 계속 조회
 */
function CBcheckIOHealthAppData() {
    if (IOHealthCheckLength == IOHealthCheckIndex) { //완료시(기존로직)
        // 59분 이전에 재조회 할 경우 저장하지 않는 오류로 추가 2020-06-29  
        var lastIndex = IOHealthAppDatas.entries.length -1;
        var lastIndexPerd = IOHealthAppDatas.entries[lastIndex].period;
        if( moment(lastIndexPerd.from).format("HH") == moment().format("HH") ) {  
            lastIndexPerd.to = moment().format('YYYY-MM-DD HH:mm:ss');  // 마지막조회시간 현재시간으로 세팅
        }
        var options = {
            path: INTF.ACT.ACT102,
            data : {
                rawData : IOHealthAppDatas
            },
            indicator : false,
            callback : function(rd){
                if (rd.rtnCd == "0000") {
                    var strgMonthlyGoal = SessUtil.getGoalList().MONTHLY_STP;
                    if(strgMonthlyGoal == 0){
                        var goalObj = {
                            "DAILY_STP" : rd.dailyGoalStep,
                            "MONTHLY_STP" : rd.monthlyGoalStep
                        }
 
                        var sessionUser =  SessUtil.getUserInfo();
                        sessionUser.activityNm = rd.activityNm;
                        sessionUser.activityCode = rd.activityCode;
 
                        SessUtil.updateUserInfo(sessionUser);
                        SessUtil.setGoalList(goalObj);                  // 목표 설정 
                    }
                    
                    MData.global("lastActivityDate", moment(rd.lastActivity).format('YYYY-MM-DD HH:mm:ss'));
                    getMonthlyStep();                   
                }
                IOHealthStepList = null; //메모리 클리어
                IOHealthAppDatas = null; //메모리 클리어
            }
        }
        MNet.httpSend(options);
        
    } else if (IOHealthAppDatas.entries.length >= 9000) { // (추가)시간별 데이터 9000개이상되면 한번 전송 하고 비우기
        var options = {
            path: INTF.ACT.ACT102,
            indicator : false,
            data : {
                rawData : IOHealthAppDatas
            },
            callback : function(rd){
                
                //전송용 rawData 재 초기화
                IOHealthAppDatas = {  //초기화
                        entries : [],
                        source : {
                            mode : 20,
                            type : 1009,
                            name : "hello",
                            product: {
                                name : "iOSHealth",
                                vender : "apple"
                            }
                        }
                    };
                
                //계속 이어간다.
                checkIOSChildActivityData( IOHealthStepList[IOHealthCheckIndex] );
            }
        }
        MNet.httpSend(options);
 
    } else if ( IOHealthStepList.length > IOHealthCheckIndex ) { //인덱스가 남았으면 계속 이어서 조회
        checkIOSChildActivityData( IOHealthStepList[IOHealthCheckIndex] );
    
    }
}

function eventCheck(){
    if(MData.global("eventInfo") != null && MData.global("eventInfo") != ""){
        var stratDate = MData.global("eventInfo").startDt.replace(/-/g,"");
        var endDate = MData.global("eventInfo").endDt.replace(/-/g,"");
        var displaySt = MData.global("eventInfo").displayStartDt;
        var dispalyEnd = MData.global("eventInfo").displayEndDt;
 
        var today = new Date();
        var yyyy = today.getFullYear();
        var mm = today.getMonth() < 9 ? "0" + (today.getMonth() + 1) : (today.getMonth() + 1); // getMonth()
        var dd  = today.getDate() < 10 ? "0" + today.getDate() : today.getDate();
        
        if(parseInt(yyyy+mm+dd) >= parseInt(stratDate) && parseInt(yyyy+mm+dd) <= parseInt(endDate)){
            $(".banner-box.event").removeClass("none");
            $(".banner-box.event p.date").text("기간 : "+ displaySt + " ~ " + dispalyEnd);
            $(".banner-box.event").on("click", function(){
                 /* 장내미생물 이벤트 페이지로 이동 */
            })
        }
    }
}

var challengeInitCb = function(receivedData){
    console.log(receivedData);
    var intestinalHealth = receivedData.intestinalHealth;
    
    checkHealthService(); //헬스 정보 체크
    // 상태값 : state
    // NONE 진행 X
    // JOIN_ING 신청 모집
    // JOIN_END 신청 마감
    // ING_NULL 미션진행중 배변 활동 미기입
    // ING_ING 미션 진행 중
    // ING_REWARD 미션 진행중 리워드 받기 가능
    // SUCCESS 미션 진행중 리워드 받기 완료
    // SUCCESS_END 완료(더이상 챌린지 표기X)
    // FAIL_END 챌린지 실패
    // FAIL_REWARD 수령 기간만료
    // JOIN_COMPLETE 신청 완료
    if( intestinalHealth.state != "NONE" && intestinalHealth.state != "SUCCESS_END" ){
        
        var _dDay = " (D"+intestinalHealth.dday+")";
        var _endDate = intestinalHealth.displayEndDate;
        var _successDate = "";
        var _displayRecuitmentDate = intestinalHealth.displayRecuitmentDate;
        if(intestinalHealth.displaySuccessDate != undefined){
            _successDate = intestinalHealth.displaySuccessDate;
        }
        
        // 서버에서 내려온 날짜 정보로 카드 일괄변경
        $("#ING_NULL_endDate").text(_endDate+_dDay);
        $("#ING_ING_endDate").text(_endDate+_dDay);
        $("#ING_REWARD_endDate").text(_endDate+_dDay);
        $("#SUCCESS_successDate").text(_successDate);
        $("#FAIL_REWARD_endDate").text(_endDate);
        $("#FAIL_END_endDate").text(_endDate+_dDay);
        $("#"+intestinalHealth.state+"_recuitmentDate").text(_displayRecuitmentDate);
        
        if(( intestinalHealth.state == "ING_ING" || intestinalHealth.state == "FAIL_END") && intestinalHealth.bowelCount > 0){
            $("#"+intestinalHealth.state).addClass("active"+intestinalHealth.bowelCount);
        }
        // 리워드 받기가 가능한 경우 리워드 받기 이벤트 추가
        if(intestinalHealth.state == "ING_REWARD"){
            $("#btnKidhealthReward").on('click',function(e){
                //한화생명 고객확인
                var sendData = SessUtil.getSecretInfo();
                if(sendData == "" || sendData == undefined) {   // 스토리지에 저장된 고객인증 데이터가 없을 경우 
 
                    /* 한화생명 고객인증(풀팝업) 페이지로 이동 */
 
                }else{
                    var options = {
                        path: INTF.MIC.MIC01,
                        data : sendData,
                        callback : intestinalHealthCheckCb
                    }
                    MNet.httpSend(options);
                }
            });
        }
        
        $("#"+intestinalHealth.state).removeClass("none");
 
    }else{
        $(".card-area.kidhealth").addClass("none");
    }
 
}
