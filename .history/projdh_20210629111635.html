/**
 * 화면명 : 이적 > 제품 스캔 안내
 * 화면 ID : TRSF0100
 * 최초 생성일자 : 
 * 
 */
 
var gbPlantCd = '';        // 선택된 플랜트 코드 값 
var gbBarcodeArr = new Array();    // 바코드 str을 담을 배열 
var gbScannedListLen = 0;
var gbTransList = [];
var gbBarcodeStr = '';        // 바코드 str 
var gbRfidObj = {};            // rfid태그로 인식된 정보 object
var gbStepNo = 2;
 
var pageInit = {
    load : function() {
        /* 첫화면 그려주기 */
        gfn_setNavigationArea();
        gfn_setPlantInfoList();
        
        /* 권한 승인 확인 -> 성공 -> 비전센서 연결 -> RFID 연결 */
        gfn_checkConnectStatus(true);
        
        // 무선스캐너 연결 
        if(GlobalInfo.SCANNER_STATUS != "Y") { exWMSetZebraScanResultListener("fn_exWMCBSetZebraScanResultListener"); }
        
        gbPlantCd = $('#btnSltdPlantNm').attr('value');
        
    },
    setEvent : function() {
        
        // 스캔 
        $('#btnStartScan').on('click',function(){
            if( $(this).hasClass('ing') ) {
                $('#btnStartScan').removeClass("ing");
                $('#btnStartScan').addClass("off");
                
                exWNStopScan(); 
//                fn_nextStopScan();
            } else     {
                if(GlobalInfo.VISION_STAT == "Y"){
                    $('#btnStartScan').removeClass("off");
                    $('#btnStartScan').addClass("ing");
                    exWNStartScan('fn_exWNCBStartScan');
                }else{
                    alert("비전센서 연결상태를 확인해주세요.",null,null,'auto');
                }
            }
        });
        
        // 이적 팝업 
        $('#btnTransSubmit').on('click',function(){
            if(Global.pageChange == true)     {fn_updateTransferList();}
            else                                {return 0;}
        });
        
        $('[id^="btnPlantNm"]').on('click',function(){
            gfn_selectPlant( $(this) );
        });
        
    }
};
 
function fn_exWNCBStartScan(obj){
//    console.log("fn_exWNCBStartScan  " + JSON.stringify(obj));
    if(obj.code == 0) {
        if(obj.resultData.bar_code != "NOREAD"){
            gbBarcode = obj.resultData.bar_code;
            fn_nextStopScan("VISION");
        }
        
//        exWNStopScan();
    } else {
        alert("바코드 다시 인식해주세요.",null,null,'auto');
    }
    
    $('#btnStartScan').removeClass("ing");
    $('#btnStartScan').addClass("off");
}
 
//무선스캐너 연결 콜백 
function fn_exWMCBSetZebraScanResultListener(obj){
//    console.log("fn_exWMCBSetZebraScanResultListener");
    
    if(obj.code == 0) {
        // 글로벌에 저장  
        gfn_util.setZebraScannerStatus("Y");
        gbBarcode = obj.resultData.barcode;
        
        fn_nextStopScan("GUN");
    } else { 
        alert("바코드 다시 인식해주세요.",null,null,'auto');
    }
}
 
/**
 * 스캔 후 동작 
 * @param type : (gun : 무선 , vision : 비전센서 )  
 * @returns
 */
function fn_nextStopScan(type){
    if(type == "VISION") {
        $('#btnStartScan').removeClass("ing");
        $('#btnStartScan').addClass("off");
    }
    
    if(gbScannedListLen < 2){
        if(gbBarcodeArr[gbBarcodeArr.length-1] == gbBarcode){
            alert("이미 스캔한 바코드입니다.",null,null,'auto');
        }else{
            gbBarcodeArr.push(gbBarcode);
//            console.log("gbBarcodeArr     "+gbBarcodeArr);
            fn_getTransferList(type);
        }
    }else{
        alert("최대 2개까지만 스캔할 수 있습니다.",null,null,'auto');
    }
    
}
 
/* 2단계 페이지 이동 - 바코드 스캔 된 경우 */
function fn_getTransferList(type){
    m.net.send( 'retrieveMwmsMoveBarcodeScan', {
            "iLang": "KR",
            "iWerks": gbPlantCd,
            "iUserid": Global.userInfo.userId,
            "iBarno": gbBarcode,         // "1200628000094",
            "iScandev" : type
    },
    function (res) {
        if(res.resultList.length > 0) {
            if(res.resultList[0].code == "NG") {
                alert(res.resultList[0].message,null,null,'auto');
                gbBarcodeArr.pop();
            }else{
                gbTransList.push(res.resultList[0]);
//                gbBarcodeArr.push(res.resultList[0].barno);
                if(gbStepNo == 2)    { drawContentArea("TRSF0200"); }
                else                    { fn_setTransferList(gbTransList); }
                
            }
        }
    });
}
 
function fn_setTransferList(list){
    $('#divScanList').empty();
    
    var scannedList = '';
    gbScannedListLen = 0;
    
    if(gbStepNo == 2)    {
        /* 2단계 페이지 이동 - 바코드 스캔만 된 경우 */
        scannedList = '<p id="pLocMsg" class="font26 mb20">'
                    +    '<span class="t-green iht">이적하시려는</span> 로케이션으로 이동<span class="t-green iht">해 주세요.</span>'
                    +'</p>';
        
        fn_beforeStartRfid();
    }else {
        /* 3단계 페이지 이동 - RFID 인식 된 경우 */
        $('#olStep li:eq(2)').attr('class',"on");
        
        scannedList = '<p id="pLocMsg" class="font26 mb20 bl">'
                    +    '<span class="t-green fw600 font36">'+ gbRfidObj.warehousenm +' '+ gbRfidObj.locnm +'</span> 로케이션에 도착<span class="t-green iht">했습니다.</span>'
                    +'</p>';
        
        $('#divCurrLoc').addClass('location');
        $('#divCurrLoc').text(gbRfidObj.warehousenm +' '+ gbRfidObj.locnm);
    }
    
    
    $.each(list, function(idx, val) {
        
        scannedList += '<div data-no="'+idx+'" class="shed-box">'
                    +        '<ul>'
                    +            '<li class="disblock"><span>바코드No.</span> <b>'+ val.barno +'</b></li>'
                    +            '<li class="disblock"><span>자재</span> <b class="ellipsis fw700">'+ val.maktx +'</b></li>'
                    +        '</ul>'
                    +        '<div class="line">'
                    +            '<ul>'
                    +                '<li><span>수량</span> <b>'+ val.qty + val.meins +'</b></li>'
                    +                '<li><span>LotNo.</span> <b>'+ val.lotno +'</b></li>'
                    +            '</ul>'
                    +        '</div>'
                    +        '<button type="button" id="btnDelete'+idx+'" class="close">닫기</button>'
                    +     '</div>';
        
        if(list.length == 1){    // 1개일 경우 비어있는 리스트 슬롯
            scannedList +=    '<div class="shed-box none">'
                        +        '<div class="scan-img mt10 mb10"></div>'
                        +        '<p>최대 2개의 보드를 스캔할 수 있습니다.</p>'
                        +    '</div>';
        }
        
        gbScannedListLen += 1;
        
        });
    $('#divScanList').html(scannedList);
    
    // 리스트 삭제 버튼 클릭 
    $('[id^="btnDelete"]').on('click',function(){
        $delBtn = $(this);
        confirm("스캔한 제품 목록을 삭제하시겠습니까?",function(){
            fn_deleteTransferList( $delBtn );
        });
    });
    
}
 
function fn_deleteTransferList( $el ){
    var index = $el.parent('div').data('no');
    index = (gbScannedListLen > 1) ? index : 0;
//    console.log("취소할 인덱스 >>>> ",index);
    
    m.net.send('updateMwmsPpPoBarcodeScanCancel',
            {
                "iLang": "KR",
                "iWerks": gbPlantCd,
                "iUserid": Global.userInfo.userId,
                "iBarno": gbBarcodeArr[index]
            },function(res){
                if(res.resultList.length > 0) {
                    if(res.resultList[0].code == "NG") {
                        alert(res.resultList[0].message,null,null,'auto');
                    }else{
                        if(gbScannedListLen > 1)    {
                            $el.parent('div').remove(); 
                            gbTransList.splice(index,1);
                            gbBarcodeArr.splice(index,1);    //인덱스길이부터 한개의 원소 지움 
                            gbScannedListLen -= 1;
                        }else {
                            drawContentArea('TRSF0100'); 
                        }
                        
                    }
                }
    });
 
}
 
// rfid 시작 전 로직 
function fn_beforeStartRfid(){
    //화면 그린 후 1초 뒤 RFID 시작
    setTimeout(function() {
        var statusRfid = setInterval(function(){ //rfid 연결됐을 경우에만 시작하기 위해 상태 interval확인  
            if(GlobalInfo.RFID_STAT == "Y"){
                clearInterval(statusRfid);
                exWNStartRfidRead('fn_exWNCBStartRfidRead');     //rfid 읽기 시작 
            } 
        }, 500);
        
    }, 1000);
}
 
//rfid 읽기 시작 콜백

        fn_getWarehouseLocation(obj.resultData.tag_value.substring(3));
    }else{                
        
    }
}

}
 
/* rfid 태그정보로 창고명,로케이션 가져오기 */
function fn_getWarehouseLocation(tag){
    m.net.send('retrieveMwmsMasterLocSearch',
            {
                "iLang": "KR",
                "iWerks": gbPlantCd,
                "iUserid": Global.userInfo.userId,
                "iLgort": '',
                "iLoc": tag
            },function(res){
                if(res.resultList.length > 0) {
                    if(res.resultList[0].code != "NG") {
//                        gbLocNm = res.resultList[0].locnm;
//                        gbLocCd = res.resultList[0].loc;
                        gbRfidObj = res.resultList[0];
                        gbStepNo = 3;
                        fn_setTransferList(gbTransList);
                    }
                }
    });
}
 
/* 3단계 페이지 이동 - RFID 인식 된 경우 */
function fn_thirdProcess(){
    $('#olStep li:eq(2)').attr('class',"on");
    var msgHtml = '';
    gbStepNo = 3;
    
    msgHtml = '<span class="t-green fw600 font36">'+ gbRfidObj.warehousenm +' '+ gbRfidObj.locnm +'</span> 로케이션에 도착<span class="t-green iht">했습니다.</span>'
    
    $('#pLocMsg').attr('class','font26 mb10 bl');
    $('#pLocMsg').html(msgHtml);
    $('#divCurrLoc').addClass('location');
    $('#divCurrLoc').text(gbRfidObj.warehousenm +' '+ gbRfidObj.locnm);
}
 
//이적 처리 
function fn_updateTransferList(){
//    var transferObj = gbTransList[0];
            
    m.net.send( 'updateMwmsMoveSetLoc', {
                "iLang": "KR",
                "iWerks": gbPlantCd,
                "iUserid": Global.userInfo.userId,
                "iLoctype": "TS",        
                "iLoc": gbRfidObj.loc,            //"7A006"
                "iBarno": gbBarcodeArr.join(',')    //"1200628000094"
            }, function (res) {
//                    console.log(JSON.stringify(res));
                    if(res.resultList[0].code == "NG") {
                    alert(res.resultList[0].message,null,null,'auto');
                }else{
                    //step1 으로 전환 
                    if(gbRfidObj.loc == "")        { toast("임시 위치에 이적 처리가 완료되었습니다."); }
                    else                            { toast("이적 처리가 완료되었습니다."); }
                        
                    drawContentArea("TRSF0100");
                }
    });
}
 
/**
 * content 클래스 영역 그려줌 + 상단 위치정보 보여줌 
 * @param page : 페이지 
 * @returns
 */
function drawContentArea(page){
    Global.pageName = page;
    if(Global.pageName == "TRSF0200"){
        $('.content').load(Global.pageName + ".html .scanbox-wrap",'', function(){ fn_setTransferList(gbTransList) }  ); 
        
        Global.pageChange = true;
    }else{
        $('.content').load(Global.pageName + ".html .scanbox-wrap" ); 
        
        gbTransList = [];
        gbBarcodeArr = [];    
        gbScannedListLen = 0;
        gbStepNo = 2;
        
        Global.pageChange = false;
        
        exWNStopRfidRead();
        exWNRfidResetTag();
    }
}
 
function fn_completeConnect(){
    exWMSetVisionSensorPhysicalButtonResultListener("fn_exWMCBSetVisionSensorPhysicalButtonResultListener");
    exWNSetUsbDetachCallback("exWNCBSetUsbDetachCallback");
}
 
 
M.onReady(function(e) {
    pageInit.load();
    pageInit.setEvent();    
});

    exWNStopScan();
    exWNStopRfidRead();
    exWNRfidResetTag();
});

    if( $('.content #divScanList').length > 0 ){ //리스트 화면일 경우 RFID 재시작 
        fn_beforeStartRfid();
    }
});
 
M.onDestroy(function(e) {
    exWNStopScan();
    exWNStopRfidRead();
    exWNRfidResetTag();
});
 
M.onPause(function(e) {
    exWNStopScan();
    exWNStopRfidRead();
    exWNRfidResetTag();
});
 
M.onResume(function(e) {
    if( $('.content #divScanList').length > 0 ){ //리스트 화면일 경우 RFID 재시작 
        fn_beforeStartRfid();
    }
});
